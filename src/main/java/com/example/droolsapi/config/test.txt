select
  count(*) as total_rows,
  count(distinct to_varchar(SRC_REC_PARTN) || ':' || to_varchar(SRC_REC_OFST)) as unique_rows,
  count(*) - count(distinct to_varchar(SRC_REC_PARTN) || ':' || to_varchar(SRC_REC_OFST)) as duplicate_rows
from <your_table>
where DW_CREAT_DTTM >= dateadd(minute, -5, current_timestamp());
