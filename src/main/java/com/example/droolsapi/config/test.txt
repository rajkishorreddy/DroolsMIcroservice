import zipfile, textwrap

jar = "/dbfs/FileStore/jars/drools-streamlit-utils-0.0.1.jar"  # adjust if your name differs

with zipfile.ZipFile(jar) as z:
    names = z.namelist()

print("Total entries:", len(names))

# Show packages present (first few levels)
pkgs = sorted({n.split("/")[0] for n in names if "/" in n})
print("\nTop-level packages:\n", pkgs[:50])

# Look for Drools/KIE
hits = [n for n in names if n.startswith(("org/kie", "org/drools", "org/mvel", "META-INF/services"))]
print(f"\nDrools/KIE-related entries found: {len(hits)}")
for n in hits[:100]:
    print(n)
