import zipfile

jar_paths = [
    "/dbfs/FileStore/jars/CommonDataObject-1.0.jar",
    "/dbfs/FileStore/jars/drools-utils-0.0.1-FAT.jar"
]

for p in jar_paths:
    try:
        with zipfile.ZipFile(p) as z:
            hits = [n for n in z.namelist() if n.endswith("MyDroolsConfig.class")]
            print("\nJAR:", p)
            print("Matches:", hits if hits else "(none)")
    except Exception as e:
        print("Error reading", p, e)
