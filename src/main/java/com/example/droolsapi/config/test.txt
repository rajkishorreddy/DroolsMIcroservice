val indvId: Long = try {
  val raw = r.getAs[Any]("INDV_ID")
  raw match {
    case null => -1L
    case n: java.lang.Long => n
    case n: java.lang.Integer => n.toLong
    case n: java.math.BigDecimal => n.longValue()
    case s: String if s.trim.nonEmpty => s.trim.toLong
    case _ => -1L
  }
} catch {
  case _: Throwable => -1L
}
